--查欄位 2 (employees之欄位)

select name 
from sys.columns  
where object_id = (select object_id 
                   from sys.tables  
                   where schema_id = 1 
                   and name = 'employees'
)



--查詢 table 及 schema name

select s.name, t.name 
from sys.tables t JOIN sys.schemas s
ON t.schema_id = s.schema_id 
where s.name = 'dbo'
