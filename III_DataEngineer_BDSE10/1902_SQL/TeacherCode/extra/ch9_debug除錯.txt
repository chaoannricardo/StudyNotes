use northwind
go

create procedure CountOrder
as
BEGIN
  declare @cnt int
  select  @cnt = count(*) from orders
  RETURN @cnt
END


--test copy 於另一視窗

BEGIN
  declare @value int, @msg  varchar(30)
  set @msg  = '訂單數: '

  exec @value = CountOrder;
  if @value > 0
    set @msg += CAST(@value as varchar)
  else
    set @msg += '沒有訂單'
  print @msg 

declare @i  as int 
set @i=1
while ( @i<=5)
begin 
select @i
set @i= @i + 1
end


END
